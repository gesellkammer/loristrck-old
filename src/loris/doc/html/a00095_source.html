352 {
353  while ( begin_partials != end_partials )
354  {
355  // do not try to add unlabeled Partials, or
356  // Partials labeled beyond 256:
357  if ( 0 != begin_partials->label() && 257 > begin_partials->label() )
358  {
359  addPartial( *begin_partials );
360  }
361  ++begin_partials;
362  }
363 }
